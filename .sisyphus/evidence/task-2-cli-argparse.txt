--- sftp-parallel --help ---
usage: sftp-parallel [-h] [--version] {upload} ...

Parallel SFTP uploader with verification

positional arguments:
  {upload}
    upload    upload a local directory to a remote host

options:
  -h, --help  show this help message and exit
  --version   show program's version number and exit

--- sftp-parallel upload --help ---
usage: sftp-parallel upload [-h] [-t N] [--verify] [--skip-existing] [-v]
                            LOCAL_DIR HOST:REMOTE_DIR

positional arguments:
  LOCAL_DIR        local directory to upload from
  HOST:REMOTE_DIR  remote destination in HOST:REMOTE_DIR format

options:
  -h, --help       show this help message and exit
  -t, --threads N  number of parallel sessions (default: 2, max: 16)
  --verify         verify uploads with checksum
  --skip-existing  skip files that exist on remote with same size
  -v, --verbose    enable debug output

--- sftp-parallel upload -t 4 /tmp/test user@host:/remote ---
